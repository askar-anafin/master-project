\section{Relevance}
\begin{frame}{Relevance of Thesis Topic}
    \begin{columns}[T]
        \begin{column}{0.6\textwidth}
            \textbf{Clinical Significance \& Motivations}
            \begin{itemize}
                \item \textbf{Global Mortality:} CVDs are the leading cause of death worldwide, causing 17.9M deaths/year.
                \item \textbf{Clinical Bottleneck:} Manual ECG interpretation is slow, subjective, and prone to fatigue.
                \item \textbf{High Variability:} Inter-observer diagnostic error exceeds 20\% in emergency departments.
            \end{itemize}
            \vspace{0.2cm}
            \textbf{Research Gaps in Computer-Aided ECG}
            \begin{itemize}
                \item \textbf{Fragile Features:} Traditional ML relies on noise-sensitive peak detection.
                \item \textbf{Spatial Gap:} Standard 1D CNNs ignore the 3D anatomical geometry of electrode placement.
            \end{itemize}
        \end{column}
        \begin{column}{0.4\textwidth}
            \begin{block}{The Research Need}
                Develop deep learning systems that model local wave morphology and spatial lead projections for real-time, explainable triage.
            \end{block}
        \end{column}
    \end{columns}
\end{frame}

                    \item \textbf{STTC:} ST/T changes
                    \item \textbf{CD:} Conduction Disturbance
                    \item \textbf{HYP:} Hypertrophy
                \end{itemize}
                Multiple diagnoses can overlap.
            \end{block}
        \end{column}

\begin{frame}{Zero-Phase Signal Preprocessing}
    \begin{columns}[T]
        \begin{column}{0.52\textwidth}
            \textbf{Artifact Denoising \& Standardization}
            \begin{itemize}
                \item \textbf{Noise Sources}: Baseline wander (<0.5 Hz) from breathing and high-frequency EMG muscle noise.
                \item \textbf{Bandpass Filter}: 4th-order Butterworth digital filter (0.5 to 50 Hz).
                \item \textbf{Zero-Phase Delay}: Forward-backward filtering (filtfilt) cancels out phase shifts and prevents wave distortion.
                \item \textbf{Normalization}: Lead-wise Z-score scaling standardizes amplitudes across records.
            \end{itemize}

\section{Models and Methods for Solving the Problem}
\begin{frame}{Baseline Models: CNN \& ViT}
    \begin{columns}[T]
        \begin{column}{0.5\textwidth}
            \begin{block}{1D Adapted ResNet-18 (CNN)}
                \begin{itemize}
                    \item \textbf{Temporal Morphology}: Convolutions capture wave shapes along the temporal axis.
                    \item \textbf{Inductive Bias}: Weight-sharing and translation invariance make CNNs highly data-efficient.
                    \item \textbf{Architecture}: Stacks four residual stages with shortcut projections to prevent gradient degradation.
                \end{itemize}
            \end{block}
        \end{column}
        \begin{column}{0.5\textwidth}
            \begin{block}{1D Vision Transformer (ViT-1D)}
                \begin{itemize}
                    \item \textbf{Sequential Rhythm}: Models long-range temporal dependencies and cardiac rhythms using self-attention.
                    \item \textbf{Tokenization}: Splits temporal sequence into 100 non-overlapping patches, prepending a \texttt{[CLS]} token.
                    \item \textbf{Constraint}: Lacks spatial and temporal locality biases, making it data-hungry and prone to overfitting.
                \end{itemize}
            \end{block}
        \end{column}
    \end{columns}
\end{frame}

% ---------------------------------------------------------
% Slide 13: Models and Methods: ST-ReGE GNN
% ---------------------------------------------------------
% Section collapsed into Models and Methods for Solving the Problem
\begin{frame}{Spatio-Temporal ReGE GNN}
    \begin{columns}[T]
        \begin{column}{0.55\textwidth}
            \textbf{Lead-Spatial Dynamic Graph}
            \begin{itemize}
                \item \textbf{Lead-Spatial Graph}: Models 12 physical leads as graph nodes to capture spatial correlations.
                \item \textbf{Shared Backbone}: Independent 1D CNN processes each lead to extract 256-D morphological features.
                \item \textbf{Learnable Adjacency}: Trainable $A = \text{Sigmoid}(W_{\text{adj}})$ dynamically discovers functional electrical pathways.
                \item \textbf{Message Passing}: Two GCN layers propagate features over the discovered pathways.
            \end{itemize}
        \end{column}
        \begin{column}{0.45\textwidth}
            \begin{block}{Tuning \& Theoretical Concept}
                \fontsize{8}{9.5}\selectfont
                \textbf{Tuning Approach:} Uses Adam optimizer ($10^{-3}$, weight decay $10^{-4}$), batch size = 32, Cosine Annealing, and validation early stopping (patience = 10).
                \vspace{0.12cm}
                
                \textbf{GCN Layer Formula:}
                \[ H^{(l+1)} = \text{BatchNorm}\left(\text{ReLU}\left(A H^{(l)} W^{(l)}\right)\right) + H^{(l)} \]
                \vspace{0.12cm}
                
                \textbf{Theoretical Value:} Traditional GNNs use physical distances. A learnable adjacency matrix $A$ learns functional lead relations directly from data.
                \vspace{0.12cm}
                
                \textbf{Analogy:} Think of a social network: instead of leads only talking to their physical neighbors, they dynamically learn to share heart abnormality 'news' with leads on opposite walls.
            \end{block}
        \end{column}
    \end{columns}
\end{frame}

% ---------------------------------------------------------
% Slide 14: Models and Methods: Spatio-Temporal Hybrid Model
% ---------------------------------------------------------
% Section collapsed into Models and Methods for Solving the Problem
\begin{frame}{Spatio-Temporal Hybrid Model}
    \begin{columns}[T]
        \begin{column}{0.55\textwidth}
            \textbf{Morphology \& Rhythm Fusion}
            \begin{itemize}
                \item \textbf{CNN Front-End}: 3 ResNet stages compress raw signals to 313 steps to extract local QRS morphology.
                \item \textbf{Squeeze-and-Excitation}: SE channel attention blocks recalibrate lead importance before temporal modeling.
                \item \textbf{Transformer Back-End}: 3-layer Encoder (8 heads, $d_{\text{model}} = 256$) models global rhythm context.
            \end{itemize}
        \end{column}
        \begin{column}{0.45\textwidth}
            \begin{block}{Synergy \& Tuning Concept}
                \fontsize{8}{9.5}\selectfont
                \textbf{Tuning \& Regularization:} Trained with 0.2 dropout, batch size = 32, Cosine Annealing, and early stopping (patience = 10) to prevent overfitting.
                \vspace{0.12cm}
                
                \textbf{Synergy Concept:} Transformers lack "locality bias" and struggle to learn raw beat shapes. Stacking a CNN front-end to extract beat shapes lets the Transformer focus purely on rhythm sequences.
                \vspace{0.12cm}
                
                \textbf{Analogy:} Like a document review team. The CNN acts as a spellchecker to verify individual beat shapes, while the Transformer acts as a reader checking overall rhythm flow.
            \end{block}
        \end{column}
    \end{columns}
\end{frame}

% ---------------------------------------------------------
% Slide 14b: Core Outcomes: ST-ReGE \& Spatio-Temporal Hybrid
% ---------------------------------------------------------
\begin{frame}{Core Outcomes: ST-ReGE \& Hybrid}
    \begin{columns}[T]
        \begin{column}{0.5\textwidth}
            \begin{block}{ST-ReGE GNN Outcomes}
                \fontsize{7.5}{9.5}\selectfont
                \begin{itemize}
                    \item \textbf{How it Works}: Models 12 leads as graph nodes and dynamically learns physical adjacency $A = \text{Sigmoid}(W_{\text{adj}})$ combined with 2 GCN propagation layers.
                    \item \textbf{What it Solves}: The spatial lead propagation gap of traditional 1D CNNs, discovering functional electrical pathways across different cardiac walls.
                    \item \textbf{Theoretical Value}: Captures joint electrical changes across leads (e.g., inferior leads II, III, aVF) instead of independent stacked channels.
                    \item \textbf{Limitations}: Assumes lead geometry fits a learnable static graph; increased model complexity compared to standard baseline convolutions.
                \end{itemize}
            \end{block}
        \end{column}
        \begin{column}{0.5\textwidth}
            \begin{block}{Spatio-Temporal Hybrid Outcomes}
                \fontsize{7.5}{9.5}\selectfont
                \begin{itemize}
                    \item \textbf{How it Works}: Stacks a 3-stage ResNet morphological front-end, SE channel lead attention block, and a 3-layer Transformer Encoder rhythm back-end.
                    \item \textbf{What it Solves}: The data-hunger and overfitting constraints of pure 1D Vision Transformers on medium-sized database splits.
                    \item \textbf{Theoretical Value}: Blends local temporal inductive bias of CNNs (beat shape details) with global rhythm sequence attention of Transformers.
                    \item \textbf{Limitations}: Higher computational memory and parameter counts; multi-head self-attention adds latency compared to simple convolutions.
                \end{itemize}
            \end{block}
        \end{column}
    \end{columns}
\end{frame}

% Slide 15: Models and Methods: Ensemble \& Boundary Calibration
% ---------------------------------------------------------
% Section collapsed into Models and Methods for Solving the Problem
\begin{frame}{Ensemble \& Boundary Calibration}
    \begin{columns}[T]
        \begin{column}{0.31\textwidth}
            {\color{rust}\fontsize{7}{9}\selectfont\sffamily\bfseries SOFT ENSEMBLING}\\[0.1cm]
            {\color{slate}\rmfamily\bfseries Joint Probability Voting}\\[0.15cm]
            \fontsize{8}{10}\selectfont Blends CNN, GNN, and Hybrid probabilities to stabilize predictions on minority classes:
            \[ P_{\text{ens}} = \frac{P_{\text{cnn}} + P_{\text{gnn}} + P_{\text{hyb}}}{3} \]
        \end{column}
        \begin{column}{0.31\textwidth}
            {\color{teal}\fontsize{7}{9}\selectfont\sffamily\bfseries CALIBRATION}\\[0.1cm]
            {\color{slate}\rmfamily\bfseries Validation Grid Search}\\[0.15cm]
            \fontsize{8}{10}\selectfont Replaces default 0.5 boundaries with class-specific thresholds:
            \fontsize{7}{9}\selectfont
            \texttt{[NORM: 0.48, MI: 0.42, STTC: 0.34, CD: 0.47, HYP: 0.39]}
        \end{column}
        \begin{column}{0.31\textwidth}
            {\color{rust}\fontsize{7}{9}\selectfont\sffamily\bfseries CLINICAL UTILITY}\\[0.1cm]
            {\color{slate}\rmfamily\bfseries On-the-Fly Tuning}\\[0.15cm]
            \fontsize{8}{10}\selectfont Allows dynamic adjustment of diagnostic sensitivity. Grid search runs on validation set in seconds on a CPU without retraining.
        \end{column}
    \end{columns}
    \vspace{0.3cm}
    \begin{block}{Why Calibration over Cost-Sensitive Loss Weighting?}
        \fontsize{8.5}{10}\selectfont
        \begin{itemize}
            \item \textbf{Preserves Rank:} Decision tuning does not alter gradients, protecting rank-ordering AUPRC.
            \item \textbf{Efficiency:} Runs in seconds on CPU, whereas retraining cost-sensitive networks takes hours.
        \end{itemize}
    \end{block}
\end{frame}

% ---------------------------------------------------------
% Slide 15b: Why Multi-Label Evaluation Metrics Matter
% ---------------------------------------------------------
\begin{frame}{Why Multi-Label Evaluation Metrics Matter}
    \begin{columns}[T]
        \begin{column}{0.55\textwidth}
            \textbf{Imbalance \& Macro Evaluation}
            \begin{itemize}
                \item \textbf{Class Imbalance}: Accuracy is highly misleading when positive cases are rare (e.g., NORM vs. HYP is $\sim$4:1). A model predicting all negative achieves 80\%+ accuracy but is clinically useless.
                \item \textbf{Macro F1-Score}: Harmonic mean of precision and recall averaged across all classes. Ensures rare but critical pathologies are weighted equally to normal cases.
                \item \textbf{AUPRC}: Focuses strictly on positive class prediction, preventing false-positive inflation from large true-negative pools.
            \end{itemize}
        \end{column}
        \begin{column}{0.45\textwidth}
            \begin{block}{Clinical Triage Trade-Offs}
                \fontsize{8.5}{10}\selectfont
                \textbf{Recall (Sensitivity):} Maximized during screening to ensure zero active pathologies (e.g., MI) are missed (minimizing False Negatives).
                \vspace{0.15cm}
                
                \textbf{Precision (Utility):} Maximized to prevent false alarms that overwhelm emergency rooms and clinical resources (minimizing False Positives).
                \vspace{0.15cm}
                
                \textbf{PhysioNet Score:} Uses clinical cost weights to penalize high-risk errors (e.g. calling MI Normal) more severely.
            \end{block}
        \end{column}
    \end{columns}
\end{frame}

\section{Conclusion}
\begin{frame}{Conclusion}
    \textbf{Summary of Completed Work Matching the Objectives:}
    \begin{enumerate}
        \item \textbf{Butterworth Filter}: Implemented zero-phase filter and Z-score scaling to remove noise.
        \item \textbf{Baseline CNN}: Built 1D ResNet-18 baseline, achieving a strong 0.7285 Macro F1-score.
        \item \textbf{Lead GNN}: Developed ST-ReGE GNN with learnable adjacency, capturing lead spatial correlations (F1: 0.7444).
        \item \textbf{ViT Evaluation}: Evaluated 1D ViT, showing it underperforms on medium databases due to inductive bias gaps.
        \item \textbf{Hybrid Model}: Built Spatio-Temporal Hybrid combining convolutional morphology with attention rhythm.
        \item \textbf{Boundary Calibration}: Calibrated decision boundaries and ensembled models to reach 0.7506 F1 and 0.9344 AUROC.
        \item \textbf{Triage Dashboard}: Built async FastAPI prototype and dashboard GUI for real-time triage demo.
    \end{enumerate}
\end{frame}
